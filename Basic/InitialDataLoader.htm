def _reader_generator(reader):
    b = reader(1024 * 1024)
    while b:
        yield b
        b = reader(1024 * 1024)


def raw_newline_count(fname):
    """
    Counts number of lines in file
    """
    f = open(fname, 'rb')
    f_gen = _reader_generator(f.raw.read)
    return sum(buf.count(b'\n') for buf in f_gen)


def hydrate_file(id_file):
    print('hydrating {}'.format(id_file))

    gzip_path = id_file+'.jsonl.gz'
    num_ids = raw_newline_count(id_file)

    with gzip.open(gzip_path, 'w') as output:
        with tqdm(total=num_ids) as pbar:
            for tweet in twarc.hydrate(id_file.open()):
                output.write(json.dumps(tweet).encode('utf8') + b"\n")
                pbar.update(1)

def post(self, *args, **kwargs):
    if not client:
        connect()

    if "data" in kwargs:
        kwargs["data"]["tweet_mode"] = self.tweet_mode

    connection_error_count = kwargs.pop('connection_error_count', 0)
    try:
        print("posting %s %s", args, kwargs)
        last_response = self.client.post(*args, timeout=(3.05, 31),
                                              **kwargs)
        return last_response
    except (ssl.SSLError, ConnectionError, ProtocolError) as e:
        connection_error_count += 1
        print("caught connection error %s on %s try", e,
                      connection_error_count)
        if (connection_errors and
                connection_error_count == self.connection_errors):
            print("received too many connection errors")
            raise e
        else:
            connect()
            kwargs['connection_error_count'] = connection_error_count
            return self.post(*args, **kwargs)

def hydrate(iterator, trim_user=False):
    """
    Pass in an iterator of tweet ids and get back an iterator for the
    decoded JSON for each corresponding tweet.
    """
    ids = []
    url = "https://api.twitter.com/1.1/statuses/lookup.json"

    # lookup 100 tweets at a time
    for tweet_id in iterator:
        tweet_id = str(tweet_id)
        tweet_id = tweet_id.strip()  # remove new line if present
        ids.append(tweet_id)
        print(ids)
        if len(ids) == 100:
            print("hydrating %s ids", len(ids))
            resp = post(url, data={
                "id": ','.join(ids),
                "include_ext_alt_text": 'true',
                "trim_user": trim_user
            })
            tweets = resp.json()
            tweets.sort(key=lambda t: t['id_str'])
            for tweet in tweets:
                yield tweet
            ids = []

    # hydrate any remaining ones
    if len(ids) > 0:
        print("hydrating %s", ids)
        resp = post(url, data={
            "id": ','.join(ids),
            "include_ext_alt_text": 'true'
        })

            yield
